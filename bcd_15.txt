
`timescale 1ns / 1ps

module binbcd_15(
    input [3:0] g,
    output reg [15:0] bcd_out
);

    reg [11:0] z;
    integer i;

    always @(*) begin
        z = {8'b0, g};

        for (i = 0; i < 4; i = i + 1) begin
            if (z[7:4] > 4)
                z[7:4] = z[7:4] + 3;
            
            z = z << 1;
        end

        bcd_out = {4'b0, 4'b0, 4'b0, z[7:4]};
        bcd_out[3:0] = z[11:8];
    end

endmodule
